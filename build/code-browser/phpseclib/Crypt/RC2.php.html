  418. 418         }

  425. 425         // of 1024 bits. It is however possible to handle effective key
  426. 426         // lengths in range 1..1024 by expanding the key and applying
  427. 427         // inverse pitable mapping to the first byte before submitting it

  439. 439         }

  442. 442         while ($i--) {
  443. 443             $l[$i] = $pitable[$l[$i + 1] ^ $l[$i + $t8]];
  444. 444         }
  445. 445 
  446. 446         // Prepare the key for mcrypt.

  451. 451     }

  503. 503      * @see Crypt_Base::encrypt()
  504. 504      * @access private
  505. 505      * @param string $in
  506. 506      * @return string
  507. 507      */
  508. 508     function _encryptBlock($in)
  509. 509     {
  510. 510         list($r0, $r1, $r2, $r3) = array_values(unpack('v*', $in));

  525. 525             $r3 |= $r3 >> 16;
  526. 526 
  527. 527             if ($j === $limit) {
  528. 528                 if ($limit === 64) {
  529. 529                     break;
  530. 530                 }
  531. 531 
  532. 532                 // Mashing round.
  533. 533                 $r0 += $keys[$r3 & 0x3F];
  534. 534                 $r1 += $keys[$r0 & 0x3F];
  535. 535                 $r2 += $keys[$r1 & 0x3F];
  536. 536                 $r3 += $keys[$r2 & 0x3F];
  537. 537                 $limit = $actions[$limit];
  538. 538             }
  539. 539         }
  540. 540 
  541. 541         return pack('vvvv', $r0, $r1, $r2, $r3);
  542. 542     }

  544. 544     /**
  545. 545      * Decrypts a block
  546. 546      *
  547. 547      * @see Crypt_Base::_decryptBlock()
  548. 548      * @see Crypt_Base::decrypt()
  549. 549      * @access private
  550. 550      * @param string $in
  551. 551      * @return string
  552. 552      */
  553. 553     function _decryptBlock($in)
  554. 554     {
  555. 555         list($r0, $r1, $r2, $r3) = array_values(unpack('v*', $in));
  556. 556         $keys = $this->keys;
  557. 557         $limit = 44;
  558. 558         $actions = array($limit => 20, 20 => 0);
  559. 559         $j = 64;
  560. 560 
  561. 561         for (;;) {
  562. 562             // R-mixing round.
  563. 563             $r3 = ($r3 | ($r3 << 16)) >> 5;
  564. 564             $r3 = ($r3 - $keys[--$j] - ((($r0 ^ $r1) & $r2) ^ $r0)) & 0xFFFF;
  565. 565             $r2 = ($r2 | ($r2 << 16)) >> 3;

  569. 569             $r0 = ($r0 | ($r0 << 16)) >> 1;

  571. 571 
  572. 572             if ($j === $limit) {
  573. 573                 if ($limit === 0) {
  574. 574                     break;
  575. 575                 }
  576. 576 
  577. 577                 // R-mashing round.
  578. 578                 $r3 = ($r3 - $keys[$r2 & 0x3F]) & 0xFFFF;
  579. 579                 $r2 = ($r2 - $keys[$r1 & 0x3F]) & 0xFFFF;
  580. 580                 $r1 = ($r1 - $keys[$r0 & 0x3F]) & 0xFFFF;
  581. 581                 $r0 = ($r0 - $keys[$r3 & 0x3F]) & 0xFFFF;
  582. 582                 $limit = $actions[$limit];
  583. 583             }
  584. 584         }
  585. 585 
  586. 586         return pack('vvvv', $r0, $r1, $r2, $r3);
  587. 587     }

  688. 688                     $r2 |= $r2 >> 16;

  690. 690                            ((($r0 ^ $r1) & $r2) ^ $r0)) & 0xFFFF) << 5;
